91  bool addStopPoint(unsigned long id, double sStop, double standingTime) {
92 
93  if(_elements.find(id) != _elements.end() && standingTime == 0) {
94  _elements[id].passed = true;
95  return true;
96  }
97  // only add if not already added
98  if(_elements.find(id) != _elements.end())
99  return false;

102  if(_sActual - sStop >= DELETE_AFTER_DISTANCE - EPS_DISTANCE)
103  return false;
104 
105  // add to list

144  void update(double actualPosition, double actualTime) {
145 
146  _sActual = actualPosition;
147 
148  // iterate over elements
149  for(auto &ke : _elements) {
150 
151  // get element
152  auto &e = ke.second;
153 
154  // ignore passed stops
155  if(e.passed)
156  continue;
157 
158  // set passed
159  if(actualTime - e.timeStartStanding >= e.standingTime - EPS_TIME)
160  e.passed = true;
161 
162  }
163 
164  // clean up
165  auto it = _elements.cbegin();
166  while(it != _elements.cend()) {
167 
168  // delete elements after passed and distance large
169  if(it->second.passed && _sActual - it->second.s >= DELETE_AFTER_DISTANCE - EPS_DISTANCE)
170  it = _elements.erase(it);
171  else
172  it++;
173 
174  }

176 
177  }
